void setup()
{

}

void song()
{
for (int note = 0; note < 51; note++)
{
int time = 1000/times[note];
tone(bzr, notes[note], time);
delay(1.25*time);
noTone(bzr);
}
}

void loop()
{
song();
delay(3000);
}
